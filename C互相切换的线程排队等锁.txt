#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

pthread_mutex_t mutex;
static char key;
void function1(){
	int i = 0;
	while(1){
		if(key == 0){
		pthread_mutex_lock(&mutex);  
		printf("function1-------------!\n");
		sleep(1);
		key = 1;
		pthread_mutex_unlock(&mutex);  
		}
	}
}

void function2(){
	int i = 0;
	while(1){
		if(key == 1){
		pthread_mutex_lock(&mutex);  
		printf("here is function2\n");
		sleep(1);
		key = 0;
		pthread_mutex_unlock(&mutex);  
		}
	}
}

int main(){
	printf("main start!\n");
	key = 0;
	pthread_mutex_init(&mutex,NULL);
	pthread_t *p1;
	pthread_t *p2;
	pthread_create(&p1,NULL,(void *)function1,NULL);
	pthread_create(&p2,NULL,(void *)function2,NULL);
	printf("end!\n");
	while(1);
	return 0;
}
